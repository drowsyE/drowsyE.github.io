시간 복잡도가 $O(n\log n)$인 대표적인 3가지 정렬 알고리즘에 대해 알아보자.

## 1. 퀵 정렬 / Quick sort

```c
#include <stdio.h>

#define swap(a, b) {int temp = (a); (a) = (b); (b) = temp;}

void quick(int arr[], int left, int right) {   
    if (left >= right) return;
    
    int lo = left+1, hi = right;
    // pivot을 왼쪽 끝으로 설정했으므로, 나중에 right와 교환한다.
    // pivot을 오른쪽 끝으로 설정한 경우, left랑 교환한다.
    int pivot = arr[left];

    while (lo <= hi) {
        while (lo <= hi && pivot >= arr[lo]) ++lo;
        while (lo <= hi && pivot < arr[hi]) --hi;    
        if (lo < hi) swap(arr[lo], arr[hi]);         
    }
    swap(pivot, arr[hi]);                           

    quick(arr, left, hi-1);    
    quick(arr, hi+1, right);
}

int main() {
    int arr[] = {5,3,1,9,6,8,2,4,7};
    int n = sizeof(arr)/sizeof(arr[0]);
    quick(arr, 0, n-1);

    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
}
```

## 2. 힙 정렬 / Heap sort

```c
#include <stdio.h>
#include <stdlib.h> // for malloc
#include <string.h> // for memcpy    

typedef struct heap {
    int* arr;
    int capacity;
    int size;
} MinHeap;

void init(MinHeap* heap, size_t n) {
    heap->capacity = n;
    heap->size = 0;
    heap->arr = (int*) calloc(n+1, sizeof(int));
}

void downheap(MinHeap* heap, int p) {
    int c = p*2;
    int key = heap->arr[p];

    while (c <= heap->size) {
        if (c != heap->size && heap->arr[c] > heap->arr[c+1]) ++c;
        if (key <= heap->arr[c]) break;

        heap->arr[p] = heap->arr[c];
        p = c;
        c = p*2;
    }
    heap->arr[p] = key;
};

int pop(MinHeap* heap) {
    if (heap->size < 1) {
        fprintf(stderr, "Pop from empty heap!\n");
        return -1;
    }

    int ret = heap->arr[1];
    heap->arr[1] = heap->arr[heap->size--];
    downheap(heap, 1);
    return ret;
}

void clear(MinHeap* heap) {
    free(heap->arr);
    heap->arr = NULL;
    heap->capacity = 0;
    heap->size = 0;
}

void heapify(MinHeap* heap, int arr[], int n) {
    memcpy(heap->arr+1, arr, n * sizeof(int));
    heap->size = n;
    for (int i = heap->size/2; i > 0; --i) {
        downheap(heap, i);
    }
}

void heapSort(int arr[], int n) {
    // build heap
    MinHeap minheap;
    init(&minheap, n);
    heapify(&minheap, arr, n);

    // pop
    while (minheap.size > 0) {
        printf("%d ", pop(&minheap));
    }

    clear(&minheap);
}

int main() {
    int arr[] = {5,3,1,9,6,8,2,4,7};
    int n = sizeof(arr)/sizeof(arr[0]);
    heapSort(arr, n);
}
```

## 3. 합병 정렬 / Merge sort

```c
#include <stdio.h>
#include <stdlib.h> // for malloc
#include <string.h> // for memcpy (optional)

void _merge_sort(int arr[], int temp[], int low, int high) {
    if (low >= high) return;

    int mid = (low + high) / 2;

    _merge_sort(arr, temp, low, mid);
    _merge_sort(arr, temp, mid+1, high);

    int i = low, j = mid+1, k = 0;
    while (i <= mid && j <= high) {
        if (arr[i] < arr[j]) {
            temp[k++] = arr[i++];
        } else {
            temp[k++] = arr[j++];
        }
    }

    // copy remaining values
    if (i <= mid) {
        while (i <= mid) {
            temp[k++] = arr[i++];
        }
    } else {
        while (j <= high) {
            temp[k++] = arr[j++];
        }
    }

    memcpy(arr + low, temp, sizeof(int) * (high-low+1));
}

void merge_sort(int arr[], size_t n) {

    // auxiliary array for storing sorted values
    int* temp = (int*) malloc(sizeof(int) * n);
    _merge_sort(arr, temp, 0, n-1);
    free(temp);
}

int main() {
    int arr[] = {5,3,1,9,6,8,2,4,7};
    int n = sizeof(arr)/sizeof(arr[0]);
    merge_sort(arr, n);

    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
}
```
